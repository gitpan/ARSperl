Field Limit Structure

; integer
{
 "min"             => integer,
 "max"             => integer
}

      or

; real
{
 "min"             => float,
 "max"             => float,
 "precision"       => integer
}

       or

; char
{
 "maxLength"       => integer,
 "menuStyle"       => "append" or "overwrite",
 "match"           => "anywhere" or "leading" or "equal",
 "charMenu"        => string,
 "pattern"         => string,
 "fullTextOptions" => "none" or "indexed"}

        or

; diary
{
 "fullTextOptions" => "none" or "indexed"}
        or
; enum and bitmask
array of strings

        or

; enum
{
 "enumLimits"      => hashref to enumLimits hash 
}
        or

undef
